import re

# Function to check password strength
def check_password_strength(password):
    criteria = [
        (len(password) >= 8, "Password should be at least 8 characters long."),
        (re.search(r'[A-Z]', password), "Password should contain at least one uppercase letter."),
        (re.search(r'[a-z]', password), "Password should contain at least one lowercase letter."),
        (re.search(r'[0-9]', password), "Password should contain at least one number."),
        (re.search(r'[@$!%*?&#]', password), "Password should contain at least one special character (@, $, !, %, *, ?, & or #).")
    ]

    # Calculate strength and collect feedback
    strength = sum(1 for check, _ in criteria if check)
    feedback = [message for check, message in criteria if not check]

    # Strength levels
    if strength == 5:
        return "Strong", feedback
    elif strength >= 3:
        return "Moderate", feedback
    else:
        return "Weak", feedback

# Input from user
def main():
    password = input("Enter your password: ")
    strength, feedback = check_password_strength(password)
    
    print(f"Password Strength: {strength}")
    if feedback:
        print("Suggestions to improve your password:")
        for suggestion in feedback:
            print(f"- {suggestion}")

if __name__ == "__main__":
    main()
